### Punctuators

```plaintext
// Operators - Assignment
    ASSIGN          // =
    PLUS_ASSIGN     // +=
    MINUS_ASSIGN    // -=
    STAR_ASSIGN     // *=
    DIV_ASSIGN      // /=
    MOD_ASSIGN      // %=
    EXP_ASSIGN      // **=
    AND_ASSIGN      // &=
    OR_ASSIGN       // |=
    XOR_ASSIGN      // ^=
    SHL_ASSIGN      // <<=
    SHR_ASSIGN      // >>=
    USHR_ASSIGN     // >>>=

// Operators - Arithmetic
    PLUS            // +
    MINUS           // -
    STAR            // *
    DIV             // / -> This is going to be challenging
    MOD             // %
    EXP             // **
    INC             // ++
    DEC             // --

// Operators - Bitwise
    AND             // &
    OR              // |
    XOR             // ^
    SHL             // <<
    SHR             // >>
    USHR            // >>>
    NOT             // ~

// Operators - Logical
    LAND            // &&
    LOR             // ||
    LNOT            // !
    NULLISH         // ??

// Operators - Comparison
    EQL             // ==
    NEQ             // !=
    STRICT_EQL      // ===
    STRICT_NEQ      // !==
    LSS             // <
    GTR             // >
    LEQ             // <=
    GEQ             // >=

// Operators - Other
    ARROW           // =>
    QUESTION        // ?
    QUESTION_DOT    // ?.
    COLON           // :
```
